# Canvas 2D 动作平台游戏轻量 QA 验证方案

## 适用范围

适用于纯前端、Vite、Canvas 2D 动作平台游戏。当前 Agent-4 只提供只读验证方案，不改源码。若项目没有源码文件，先由 Agent-0 或实现 Agent 补齐源码后再执行以下命令。

## 0. 终端 UTF-8 初始化

```powershell
[Console]::OutputEncoding = [System.Text.Encoding]::UTF8
$OutputEncoding = [System.Text.Encoding]::UTF8
$env:PYTHONIOENCODING = "utf-8"
$ErrorActionPreference = "Stop"
```

## 1. 静态检查

```powershell
[Console]::OutputEncoding = [System.Text.Encoding]::UTF8
$OutputEncoding = [System.Text.Encoding]::UTF8
$ErrorActionPreference = "Stop"

if (Test-Path package.json) {
  if (Test-Path package-lock.json) {
    npm ci
  } elseif (Test-Path pnpm-lock.yaml) {
    corepack enable
    pnpm install --frozen-lockfile
  } elseif (Test-Path yarn.lock) {
    corepack enable
    yarn install --immutable
  } else {
    npm install
  }

  npm run lint --if-present
  npm run typecheck --if-present
  npm run test --if-present -- --run
  npm run build --if-present
} elseif (Test-Path index.html) {
  Get-ChildItem -Recurse -Filter *.js | ForEach-Object {
    node --check $_.FullName
  }
} else {
  throw "未发现 package.json 或 index.html，无法执行前端静态检查。"
}
```

验收断言：

- 依赖安装成功。
- `lint`、`typecheck`、`test` 如存在必须通过。
- Vite 项目必须至少通过 `npm run build --if-present`。
- 纯静态项目必须通过所有 `*.js` 的 `node --check`。

## 2. 启动本地服务

Vite 项目优先：

```powershell
[Console]::OutputEncoding = [System.Text.Encoding]::UTF8
$OutputEncoding = [System.Text.Encoding]::UTF8
$ErrorActionPreference = "Stop"

$port = 5173
$root = (Get-Location).Path
$server = Start-Process -FilePath "npm.cmd" `
  -ArgumentList @("run", "dev", "--", "--host", "127.0.0.1", "--port", "$port") `
  -WorkingDirectory $root `
  -WindowStyle Hidden `
  -PassThru

try {
  $ok = $false
  foreach ($i in 1..30) {
    try {
      Invoke-RestMethod "http://127.0.0.1:$port" -TimeoutSec 2 | Out-Null
      $ok = $true
      break
    } catch {
      Start-Sleep -Seconds 1
    }
  }
  if (-not $ok) { throw "服务未在 30 秒内启动：http://127.0.0.1:$port" }
  "服务已启动：http://127.0.0.1:$port"
} catch {
  Stop-Process -Id $server.Id -Force -ErrorAction SilentlyContinue
  throw
}
```

纯静态项目 fallback：

```powershell
npx -y http-server . -p 5173 -a 127.0.0.1
```

验收断言：

- `http://127.0.0.1:5173` 可访问。
- 首页 HTTP 成功，无 404。
- 页面加载后浏览器控制台无 `error`，无 `pageerror`。

## 3. Playwright 浏览器自动检查

在服务已启动后执行。脚本写入系统临时目录，不写项目源码。

```powershell
[Console]::OutputEncoding = [System.Text.Encoding]::UTF8
$OutputEncoding = [System.Text.Encoding]::UTF8
$ErrorActionPreference = "Stop"
$env:QA_URL = "http://127.0.0.1:5173"

$qaScript = Join-Path $env:TEMP "canvas-platform-qa.js"
@'
const { chromium } = require("playwright");

const url = process.env.QA_URL || "http://127.0.0.1:5173";
const strict = process.env.QA_STRICT === "1";
const errors = [];
const warnings = [];

function assert(condition, message) {
  if (!condition) throw new Error(message);
}

async function maybeClickText(page, pattern) {
  const candidates = [
    page.getByRole("button", { name: pattern }).first(),
    page.getByRole("link", { name: pattern }).first(),
    page.getByText(pattern).first()
  ];
  for (const locator of candidates) {
    try {
      if (await locator.count()) {
        await locator.click({ timeout: 1500 });
        return true;
      }
    } catch {}
  }
  return false;
}

async function canvasFrame(page) {
  return await page.evaluate(() => {
    const canvas = document.querySelector("canvas");
    if (!canvas) return null;
    const ctx = canvas.getContext("2d");
    if (!ctx) return null;
    const w = Math.min(canvas.width || canvas.clientWidth, 96);
    const h = Math.min(canvas.height || canvas.clientHeight, 96);
    const data = ctx.getImageData(0, 0, Math.max(1, w), Math.max(1, h)).data;
    let nonBlank = 0;
    let checksum = 0;
    for (let i = 0; i < data.length; i += 4) {
      const rgba = data[i] + data[i + 1] + data[i + 2] + data[i + 3];
      checksum = (checksum + rgba * (i + 1)) % 1000000007;
      if (data[i + 3] > 0 && (data[i] || data[i + 1] || data[i + 2])) nonBlank++;
    }
    return {
      width: canvas.width || canvas.clientWidth,
      height: canvas.height || canvas.clientHeight,
      nonBlank,
      checksum,
      dataUrlHead: canvas.toDataURL("image/png").slice(0, 1200)
    };
  });
}

(async () => {
  const browser = await chromium.launch({ headless: true });
  const page = await browser.newPage({ viewport: { width: 1280, height: 720 } });

  page.on("console", msg => {
    if (msg.type() === "error") errors.push(msg.text());
  });
  page.on("pageerror", err => errors.push(err.message));

  await page.goto(url, { waitUntil: "networkidle", timeout: 30000 });

  const title = await page.title();
  assert(typeof title === "string", "页面 title 读取失败");

  const canvas = page.locator("canvas").first();
  await canvas.waitFor({ state: "visible", timeout: 10000 });
  const box = await canvas.boundingBox();
  assert(box && box.width >= 320 && box.height >= 180, "Canvas 尺寸过小或不可见");

  const initial = await canvasFrame(page);
  assert(initial && initial.width >= 320 && initial.height >= 180, "Canvas 内部尺寸异常");
  assert(initial.nonBlank > 20, "Canvas 画面疑似空白");

  const clickedStart = await maybeClickText(page, /开始|Start|Play|进入|继续/i);
  if (!clickedStart) {
    await page.keyboard.press("Enter");
    await page.keyboard.press("Space");
  }
  await page.waitForTimeout(800);
  assert(errors.length === 0, `开始游戏后出现错误：${errors.join(" | ")}`);

  const beforeMove = await canvasFrame(page);
  await page.keyboard.down("ArrowRight");
  await page.waitForTimeout(450);
  await page.keyboard.up("ArrowRight");
  await page.keyboard.press("ArrowLeft");
  await page.waitForTimeout(250);
  const afterMove = await canvasFrame(page);
  assert(beforeMove && afterMove && beforeMove.dataUrlHead !== afterMove.dataUrlHead, "移动输入后画面未变化");

  const beforeAttack = await canvasFrame(page);
  for (const key of ["KeyJ", "KeyK", "Space"]) {
    await page.keyboard.press(key);
    await page.waitForTimeout(180);
  }
  const afterAttack = await canvasFrame(page);
  assert(beforeAttack && afterAttack && beforeAttack.dataUrlHead !== afterAttack.dataUrlHead, "攻击输入后画面未变化");
  assert(errors.length === 0, `移动或攻击后出现错误：${errors.join(" | ")}`);

  const beforeText = await page.locator("body").innerText().catch(() => "");
  const hookResult = await page.evaluate(() => {
    const w = window;
    if (typeof w.__qaCompleteLevel === "function") {
      w.__qaCompleteLevel();
      return "window.__qaCompleteLevel";
    }
    if (typeof w.__qaNextChapter === "function") {
      w.__qaNextChapter();
      return "window.__qaNextChapter";
    }
    if (w.game && typeof w.game.completeLevel === "function") {
      w.game.completeLevel();
      return "window.game.completeLevel";
    }
    return "";
  });
  if (!hookResult) {
    await page.keyboard.press("KeyN");
    await page.keyboard.press("Enter");
    warnings.push("未发现通关测试钩子，仅尝试 KeyN/Enter；如需严格验收请暴露 window.__qaCompleteLevel 或 window.__qaNextChapter。");
  }
  await page.waitForTimeout(1000);

  const afterText = await page.locator("body").innerText().catch(() => "");
  const passPattern = /第\s*2|章节\s*2|第.*关|下一章|通关|胜利|完成|Chapter\s*2|Level\s*2|Victory|Clear|Complete/i;
  const chapterChanged = passPattern.test(afterText) || beforeText !== afterText || Boolean(hookResult);
  if (strict) {
    assert(chapterChanged, "章节切换或通关断言未满足");
  } else if (!chapterChanged) {
    warnings.push("章节切换或通关未形成可观察断言，建议补充 DOM 文案、路由、状态文本或 QA hook。");
  }

  assert(errors.length === 0, `存在浏览器错误：${errors.join(" | ")}`);

  console.log(JSON.stringify({
    ok: true,
    url,
    canvas: { width: initial.width, height: initial.height, nonBlank: initial.nonBlank },
    start: clickedStart ? "clicked" : "keyboard-fallback",
    move: "passed",
    attack: "passed",
    chapterOrClear: chapterChanged ? "passed-or-hooked" : "warning",
    warnings
  }, null, 2));

  await browser.close();
})().catch(async err => {
  console.error(err);
  process.exit(1);
});
'@ | Set-Content -Path $qaScript -Encoding UTF8

npx -y -p playwright node $qaScript
```

严格通关验收：

```powershell
$env:QA_STRICT = "1"
npx -y -p playwright node $qaScript
```

自动检查验收断言：

- 页面可打开，加载完成。
- 有可见 Canvas，尺寸不小于 `320x180`。
- Canvas 像素非空白。
- 点击“开始”或键盘 fallback 后无浏览器错误。
- `ArrowRight`、`ArrowLeft` 后 Canvas 画面变化。
- `J/K/Space` 攻击后 Canvas 画面变化。
- 章节切换或通关有可观察信号：DOM 文案变化、关卡文字变化、Victory/Clear/Complete 文案、路由变化，或 `window.__qaCompleteLevel` / `window.__qaNextChapter` 测试钩子生效。

## 4. 人工补充验收

自动检查通过后，Agent-0 用浏览器补 2 分钟人工验收：

- 开始按钮：能进入游戏主循环，按钮状态或菜单消失。
- 移动：左右移动、跳跃、平台碰撞正常，不穿墙、不掉出视口后无恢复。
- 攻击：攻击动画明显，命中敌人后敌人受击、扣血、死亡或得分变化。
- 章节切换：到达终点、击败 Boss 或触发门后进入下一章。
- 通关：最终关显示胜利、结算或返回菜单，不报错不卡死。

## 5. 建议的源码侧可测试性钩子

如后续允许实现 Agent 小改源码，建议只在开发环境暴露：

```js
if (import.meta?.env?.DEV) {
  window.__qaNextChapter = () => game.nextChapter();
  window.__qaCompleteLevel = () => game.completeLevel();
  window.__qaGetState = () => ({
    chapter: game.chapter,
    playerX: game.player.x,
    playerHp: game.player.hp,
    enemiesAlive: game.enemies.filter(e => !e.dead).length
  });
}
```

对应硬断言：

- 开始后 `state.running === true`。
- 移动后 `playerX` 变化。
- 攻击命中后 `enemiesAlive` 或敌人 HP 变化。
- 通关后 `chapter` 递增或 `state.victory === true`。
